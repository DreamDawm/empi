## 患者信息表

```sql
create table im_patient
(
    patient_id             varchar(50)                        not null comment '源系统患者id'
        primary key,
    person_source_value    varchar(50)                        not null comment '患者empi',
    person_name            varchar(100)                       not null comment '患者姓名',
    year_of_birth          int                                null comment '生日年',
    month_of_birth         int                                null comment '生日月',
    day_of_birth           int                                null comment '生日日',
    time_of_birth          varchar(20)                        null comment '生日时间',
    gender_source_value    varchar(100)                       null comment '性别名称',
    race_source_value      varchar(100)                       null comment '民族名称',
    ethnicity_source_value varchar(20)                        null comment '人种名称',
    birthday               datetime                           null comment '出生日期',
    location               varchar(2000)                      null comment '住址',
    contacts               varchar(200)                       null comment '联系人',
    contacts_number        varchar(50)                        null comment '联系人电话',
    relationship           varchar(50)                        null comment '联系人关系',
    card_id                varchar(50)                        null comment '身份证号码',
    payer_plan_card_id     varchar(100)                       null comment '医保卡id',
    payer_plan_card_type   varchar(50)                        null comment '医保卡类型',
    person_career          varchar(100)                       null comment '患者职业',
    marital_status         varchar(20)                        null comment '结婚状态名称',
    work_unit              varchar(200)                       null comment '工作单位',
    degree_of_education    varchar(50)                        null comment '学历',
    height                 varchar(20)                        null comment '身高',
    weight                 varchar(20)                        null comment '体重',
    province               varchar(100)                       null comment '省名称',
    city                   varchar(100)                       null comment '市名称',
    area                   varchar(100)                       null comment '区名称',
    tapt_no                varchar(100)                       null comment '病案号',
    abo                    varchar(20)                        null comment 'ABO血型名称',
    rh                     varchar(10)                        null comment 'RH血型名称',
    identity_card_num      varchar(50)                        null comment '身份证号码',
    first_diagnosis_age    int                                null comment '首诊年龄',
    gender                 varchar(50)                        null comment '性别',
    phone                  varchar(50)                        null comment '患者电话',
    hospital_id            int                                null comment '医院ID',
    frist_visit_date       datetime                           null comment '首次就诊时间',
    source_table           varchar(50)                        null comment '来源表',
    source_id              int                                null comment '来源主键id',
    is_etl                 int      default 0                 null comment '是否同步',
    data_createtime        datetime default CURRENT_TIMESTAMP null comment '数据创建时间',
    data_updatetime        datetime default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP comment '数据更新时间'
)
    comment '患者信息表';
```
